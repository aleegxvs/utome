async function loadChildren() {
  const snap = await db.collection("children").where("userId","==",state.user.uid).get();
  state.children = snap.docs.map(d => ({id:d.id,...d.data()}));
  renderChildren();
  if (state.children.length && !state.activeChildId) {
    selectChild(state.children[0].id);
  }
}

function renderChildren() {
  const list = document.getElementById("children-list");
  list.innerHTML = "";
  if (state.children.length === 0) {
    list.innerHTML = '<div style="color:var(--muted);font-size:13px;padding:10px;text-align:center;">Nenhuma criança</div>';
    return;
  }
  state.children.forEach(c => {
    const div = document.createElement("div");
    div.className = "child-item" + (c.id === state.activeChildId ? " active" : "");
    div.innerHTML = `<div>${c.name}<small>ID: ${c.deviceId}</small></div>`;
    div.addEventListener("click", () => selectChild(c.id));
    list.appendChild(div);
  });
}

document.getElementById("add-child-btn").addEventListener("click", async () => {

  state.children.push({id:docRef.id,name,deviceId,userId:state.user.uid});
  renderChildren();
  selectChild(docRef.id);
});

async function selectChild(childId) {
  state.activeChildId = childId;
  renderChildren();
  const snap = await db.collection("tasks").where("childId","==",childId).get();
  state.tasks = snap.docs.map(d => ({id:d.id,...d.data()}));
  state.tasks.sort((a,b) => a.time.localeCompare(b.time));
  renderRoutine();
}

  label.style.marginBottom = "6px";
  label.textContent = "Dias da nova tarefa:";
  const dayToggleWrap = document.getElementById("day-toggle-wrap");
  dayToggleWrap.appendChild(label);

function renderTaskList() {
  const list = document.getElementById("task-list");
  if (!state.tasks.length) {
    list.innerHTML = `<div class="empty-state" style="padding:30px 10px;"><div class="icon">${ei("clipboard",64)}</div><h3>Nenhuma tarefa</h3><p>Adicione a primeira tarefa abaixo.</p></div>`;
    return;
  }
  list.innerHTML = "";
  state.tasks.forEach(t => {
    const days = t.days.map(d => DAY_LABELS[d]).join(", ");
    const card = document.createElement("div");
    card.className = "task-card";
    card.innerHTML = `<div class="task-heart ${t.color}">${emojiToIcon(t.emoji,18)}</div><div class="task-info"><div class="t-time">${t.time} · ${t.name}</div><div class="task-days">${days}</div></div><div class="task-actions"><button title="Remover" data-id="${t.id}">${ei("wastebasket",18)}</button></div>`;
    card.querySelector("button").addEventListener("click", () => deleteTask(t.id));
    list.appendChild(card);
  });
}

async function addTask() {
  const time = document.getElementById("nt-time").value;
  const name = document.getElementById("nt-name").value.trim();
  const emoji = document.getElementById("nt-emoji").value.trim() || "😊";
  const color = document.getElementById("nt-color").value;

  if (!time || !name) return;
  if (!newTaskDays.length) {alert("Selecione ao menos um dia.");return;}

  await db.collection("tasks").add({childId:state.activeChildId,time,name,emoji,color,days:newTaskDays,active:true});

  document.getElementById("nt-time").value = "08:00";
  document.getElementById("nt-name").value = "";
  document.getElementById("nt-emoji").value = "😊";
  document.getElementById("nt-color").value = "blue";
  newTaskDays = [0,1,2,3,4,5,6];

  await selectChild(state.activeChildId);
}

async function deleteTask(taskId) {
  await db.collection("tasks").doc(taskId).delete();
  state.tasks = state.tasks.filter(t => t.id !== taskId);
  renderRoutine();
}
