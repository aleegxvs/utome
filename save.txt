function sanitizeText(value, maxLen = 50) {
  return DOMPurify.sanitize(String(value || "").trim(), { ALLOWED_TAGS: [], ALLOWED_ATTR: [] }).slice(0, maxLen);
}

function validateText(value, { min = 1, max = 50, label = "Campo" } = {}) {
  const v = sanitizeText(value, max);
  if (v.length < min) return { ok: false, error: `${label} é obrigatório`, value: v };
  if (v.length > max) return { ok: false, error: `${label} muito longo`, value: v };
  return { ok: true, value: v };
}

function validateEmail(raw) {
  const v = sanitizeText(raw, 120);
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v)) return { ok: false, error: "E-mail inválido" };
  return { ok: true, value: v };
}

  return { ok: true };
}

function validateImageFile(file) {
  if (!file) return { ok: false, error: "Nenhum arquivo selecionado" };
  if (!UTOME_SEC.LIMITS.ALLOWED_IMAGE_TYPES.includes(file.type)) {
    return { ok: false, error: "Envie JPG, PNG ou WEBP." };
  }
  if (file.size > UTOME_SEC.LIMITS.MAX_UPLOAD_BYTES) {
    return { ok: false, error: "Imagem deve ter no máximo 500KB." };
  }
  return { ok: true };
}

function validateBase64Image(dataUrl) {
  if (!dataUrl || !dataUrl.startsWith("data:image/")) return { ok: false, error: "Formato de imagem inválido" };
  const approxBytes = Math.ceil((dataUrl.length - dataUrl.indexOf(",") - 1) * 0.75);
  if (approxBytes > UTOME_SEC.LIMITS.MAX_UPLOAD_BYTES) return { ok: false, error: "Imagem muito grande (máx. 500KB)." };
  return { ok: true };
}

function getAuthErrorMsg() {
  return "Credenciais inválidas ou erro de autenticação.";
}

function checkRateLimit(key) {
  const cfg = UTOME_SEC.RATE[key];
  if (!cfg) return { allowed: true };
  const store = JSON.parse(localStorage.getItem("utome_rate") || "{}");
  const now = Date.now();
  const entry = store[key] || { attempts: [], blockedUntil: 0 };
  if (entry.blockedUntil > now) return { allowed: false, retryAfter: entry.blockedUntil - now };
  entry.attempts = entry.attempts.filter(t => now - t < cfg.windowMs);
  if (entry.attempts.length >= cfg.max) {
    entry.blockedUntil = now + cfg.windowMs;
    store[key] = entry;
    localStorage.setItem("utome_rate", JSON.stringify(store));
    logAudit("RATE_LIMIT_HIT", { key }, "WARNING");
    return { allowed: false };
  }
  entry.attempts.push(now);
  store[key] = entry;
  localStorage.setItem("utome_rate", JSON.stringify(store));
  return { allowed: true };
}

function recordAction(action) {
  const now = Date.now();
  UTOME_SEC._anomaly.actions.push({ action, ts: now });
  UTOME_SEC._anomaly.actions = UTOME_SEC._anomaly.actions.filter(a => now - a.ts < 60000);
  if (UTOME_SEC._anomaly.actions.length > 40) {
    logAudit("ANOMALY_HIGH_ACTIVITY", { count: UTOME_SEC._anomaly.actions.length, action }, "CRITICAL");
  }
}

function recordLoginFailure() {
  UTOME_SEC._anomaly.loginFailures++;
  if (UTOME_SEC._anomaly.loginFailures >= 5) {
    logAudit("ANOMALY_LOGIN_FAILURES", { count: UTOME_SEC._anomaly.loginFailures }, "WARNING");
    UTOME_SEC._anomaly.loginFailures = 0;
  }
}

function recordUploadRejection() {
  UTOME_SEC._anomaly.uploadRejections++;
  if (UTOME_SEC._anomaly.uploadRejections >= 3) {
    logAudit("ANOMALY_UPLOAD_SPIKE", { count: UTOME_SEC._anomaly.uploadRejections }, "WARNING");
    UTOME_SEC._anomaly.uploadRejections = 0;
  }
}

function enforceResourceLimit(type, currentCount) {
  if (type === "children" && currentCount >= UTOME_SEC.LIMITS.MAX_CHILDREN) {
    return { ok: false, error: `Limite de ${UTOME_SEC.LIMITS.MAX_CHILDREN} crianças atingido.` };
  }
  if (type === "tasks" && currentCount >= UTOME_SEC.LIMITS.MAX_TASKS_PER_CHILD) {
    return { ok: false, error: `Limite de ${UTOME_SEC.LIMITS.MAX_TASKS_PER_CHILD} tarefas por criança.` };
  }
  return { ok: true };
}

async function initSessionSecurity(user) {
  const sessionId = crypto.randomUUID();
  sessionStorage.setItem("utome_session_id", sessionId);
  try { await user.getIdToken(true); } catch (_) {}
  try {
    await db.collection("users").doc(user.uid).update({
      lastSessionId: sessionId,
      lastLoginAt: firebase.firestore.FieldValue.serverTimestamp()
    });
  } catch (_) {}
  logAudit("SESSION_INIT", { sessionId });
}

function clearSessionSecurity() {
  sessionStorage.removeItem("utome_session_id");
  UTOME_SEC._anomaly = { actions: [], loginFailures: 0, uploadRejections: 0 };
}

async function withGuard(btn, rateKey, actionName, fn) {
  const rl = checkRateLimit(rateKey);
  if (!rl.allowed) {
    alert("Muitas tentativas. Aguarde alguns minutos e tente novamente.");
    return;
  }
  recordAction(actionName);
  if (btn) btn.disabled = true;
  try { return await fn(); }
  finally { if (btn) btn.disabled = false; }
}

async function handleMfaSignIn(resolver) {
  mfaResolver = resolver;
  const hint = resolver.hints[0];
  const phoneAuthProvider = new firebase.auth.PhoneAuthProvider();
  mfaVerificationId = await phoneAuthProvider.verifyPhoneNumber(
    { multiFactorHint: hint, session: resolver.session },
    getMfaRecaptcha()
  );
  openMfaModal("Verificação em 2 etapas", "Digite o código SMS enviado ao seu telefone cadastrado.");
}

      updateMfaUI();

    }
  } catch (e) {
    logAudit("MFA_FAILED", {}, "WARNING");
    alert(getAuthErrorMsg());

  }

  const errBox = document.getElementById("auth-error");
  errBox.style.display = "none";
  const emailVal = validateEmail(document.getElementById("auth-email").value);
  const password = document.getElementById("auth-password").value;
  if (!emailVal.ok) { errBox.textContent = emailVal.error; errBox.style.display = "block"; return; }
  if (!password) { errBox.textContent = "Preencha todos os campos"; errBox.style.display = "block"; return; }
  const rateKey = authMode === "login" ? "login" : "register";
  const rl = checkRateLimit(rateKey);
  if (!rl.allowed) { errBox.textContent = getAuthErrorMsg(); errBox.style.display = "block"; return; }
  setLoading(true);
  recordAction(authMode === "login" ? "LOGIN_ATTEMPT" : "REGISTER_ATTEMPT");
  try {
    if (authMode === "login") {
      const pwVal = validatePassword(password);
      if (!pwVal.ok) throw new Error("invalid");
      await auth.signInWithEmailAndPassword(emailVal.value, password);
      logAudit("LOGIN_SUCCESS", { method: "email" });
    } else {
      const nameVal = validateText(document.getElementById("reg-name").value, { max: UTOME_SEC.LIMITS.MAX_NAME, label: "Nome" });
      const pwVal = validatePassword(password);
      if (!nameVal.ok || !pwVal.ok) throw new Error("invalid");
      const res = await auth.createUserWithEmailAndPassword(emailVal.value, password);
      await db.collection("users").doc(res.user.uid).set({ name: nameVal.value, email: emailVal.value });
      logAudit("REGISTER_SUCCESS", { method: "email" });
    }
  } catch (e) {
    if (e.code === "auth/multi-factor-auth-required") {
      try {
        await handleMfaSignIn(e.resolver);
        return;
      } catch (_) {
        recordLoginFailure();
      }
    } else {
      recordLoginFailure();
    }
    logAudit("LOGIN_FAILED", { method: "email" }, "WARNING");
    errBox.textContent = getAuthErrorMsg();
    errBox.style.display = "block";
    setLoading(false);
  }
});

auth.onAuthStateChanged(async (user) => {
  if (user) {
    try {
      const doc = await db.collection("users").doc(user.uid).get();
      const data = doc.exists ? doc.data() : {};
      const name = data.name || user.displayName || "Usuário";
      if (!doc.exists) {
        await db.collection("users").doc(user.uid).set({ name, email: user.email || "" });
      }
      state.user = { uid: user.uid, name, email: user.email || "", nick: data.nick || "", photoURL: data.photoURL || null, ...data };
      if (!sessionStorage.getItem("utome_session_id")) await initSessionSecurity(user);
      closeMfaModal();
      setLoading(false);
      firebaseResolved = true;
      tryHideSplash();
      showApp();
    } catch (e) {
      console.error("[UTOME] Falha ao carregar dados do usuário após login:", e);
      firebaseResolved = true;
      tryHideSplash();
      setLoading(false);
      const errBox = document.getElementById("auth-error");
      if (errBox) {
        errBox.textContent = "Não foi possível carregar sua conta. Tente novamente em alguns instantes.";
        errBox.style.display = "block";
      }
      showAuth();
    }
  } else {
    state.user = null;
    clearSessionSecurity();
    if (taskUnsubscribe) { taskUnsubscribe(); taskUnsubscribe = null; }
    firebaseResolved = true;
    tryHideSplash();
    if (!onboardingDone) {
      document.getElementById("onboarding").style.display = "block";
    } else {
      showAuth();
    }
  }
});

async function deleteChild(childId, childName) {
  const confirmed = await showConfirmModal(
    `Remover "${childName}"?`,
    "Todas as tarefas desta criança também serão deletadas. Esta ação não pode ser desfeita."
  );
  if (!confirmed) return;
  const taskSnap = await db.collection("tasks").where("childId", "==", childId).get();
  const batch = db.batch();
  taskSnap.docs.forEach(doc => batch.delete(doc.ref));
  batch.delete(db.collection("children").doc(childId));
  await batch.commit();
  logAudit("DELETE_CHILD", { childId });
  state.children = state.children.filter(c => c.id !== childId);
  if (state.activeChildId === childId) {
    state.activeChildId = null;
    state.tasks = [];
  }
  renderChildren();
  if (state.activeChildId) {
    selectChild(state.activeChildId);
  } else {
    document.getElementById("routine-panel").innerHTML = '<div class="empty-state"><div class="icon"><img src="https://img.icons8.com/ios-filled/64/FF7A00/target.png"></div><h3>Selecione uma criança</h3><p>Escolha uma criança na esquerda para configurar sua rotina.</p></div>';
  }
}

  state.activeChildId = childId;

    });
}

function renderTaskList(prevIds, newIds) {
  const list = document.getElementById("task-list");
  if (!list) return;
  const todayIdx = getTodayIdx();
  
  let filteredTasks = state.tasks;
  
  if (taskView === "today") {
    filteredTasks = state.tasks.filter(t => (t.days || []).includes(todayIdx));
    
    if (!filteredTasks.length) {
      list.innerHTML = `
        <div class="no-tasks-today">
          <div class="icon"><img src="https://img.icons8.com/ios-filled/48/FF7A00/sun.png"></div>
          <p>Nenhuma tarefa para hoje.<br>Que tal curtir o dia livre?</p>
        </div>
      `;
      return;
    }
    
    // F: Next task card - find next upcoming task
    const now = new Date();
    const nowStr = now.getHours().toString().padStart(2,"0") + ":" + now.getMinutes().toString().padStart(2,"0");
    const upcoming = filteredTasks.find(t => (t.time || "00:00") >= nowStr);
    const nextTask = upcoming || filteredTasks[filteredTasks.length - 1];
    
    // Render next task hero card
    const heroDiv = document.createElement("div");
    heroDiv.className = "next-task-card";
    const heroIcon = SUGGESTED_ICONS.find(ic => ic.name === (nextTask.emoji || "star")) || SUGGESTED_ICONS[16];
    const heroIconSrc = heroIcon ? heroIcon.src : "https://img.icons8.com/ios-filled/32/star.png";
    heroDiv.innerHTML = `
      <div class="ntc-icon"><img src="${heroIconSrc}" width="28" height="28" style="display:block;filter:brightness(0) invert(1);"></div>
      <div class="ntc-info">
        <div class="ntc-label">${upcoming ? "Próxima tarefa" : "Última tarefa do dia"}</div>
        <div class="ntc-time">${esc(nextTask.time || "--:--")}</div>
        <div class="ntc-name">${esc(nextTask.name)}</div>
      </div>
    `;
    list.innerHTML = "";
    list.appendChild(heroDiv);
    
    // Remaining tasks (all, below hero)
    filteredTasks.forEach(t => {
      const card = createTaskCard(t, todayIdx, prevIds, newIds);
      list.appendChild(card);
    });
    
  } else {
    // Week view
    if (!state.tasks.length) {
      list.innerHTML = '<div class="empty-state" style="padding:30px 10px;"><div class="icon"><img src="https://img.icons8.com/ios-filled/48/FF7A00/task-planning.png"></div><h3>Nenhuma tarefa</h3><p>Adicione a primeira tarefa abaixo.</p></div>';
      return;
    }
    list.innerHTML = "";
    state.tasks.forEach(t => {
      const card = createTaskCard(t, todayIdx, prevIds, newIds);
      list.appendChild(card);
    });
  }
}

// I: Create task card with animation support
function createTaskCard(t, todayIdx, prevIds, newIds) {
  const card = document.createElement("div");
  card.className = "task-card";
  card.dataset.id = t.id;
  
  const dayPills = (t.days || []).map(d =>
    `<span class="task-day-pill${d === todayIdx ? " today" : ""}">${DAY_LABELS[d]}</span>`
  ).join("");
  const iconObj = SUGGESTED_ICONS.find(ic => ic.name === (t.emoji || "smile")) || SUGGESTED_ICONS[0];
  const iconSrc = iconObj ? iconObj.src : `https://img.icons8.com/ios-filled/32/smile.png`;
  card.innerHTML = `
    <div class="task-heart ${esc(t.color)}"><img src="${iconSrc}" width="22" height="22" style="display:block;filter:brightness(0) invert(1);"></div>
    <div class="task-info">
      <div class="t-hour">${esc(t.time)}</div>
      <div class="t-name">${esc(t.name)}</div>
      <div class="task-days">${dayPills}</div>
    </div>
    <div class="task-actions"><button title="Remover" data-id="${esc(t.id)}"><img src="https://img.icons8.com/ios-filled/18/8B889C/trash.png" style="vertical-align:middle;"></button></div>
  `;
  card.querySelector("button").addEventListener("click", () => deleteTask(t.id, t.name));
  
  // I: Enter animation for new tasks
  if (prevIds && newIds && !prevIds.has(t.id)) {
    card.classList.add("entering");
  }
  
  return card;
}

async function addTask() {
  await withGuard(document.getElementById("nt-save"), "createTask", "CREATE_TASK", async () => {
    const limit = enforceResourceLimit("tasks", state.tasks.length);
    if (!limit.ok) { alert(limit.error); logAudit("RESOURCE_LIMIT", { type: "tasks" }, "WARNING"); return; }
    const time = document.getElementById("nt-time").value;
    const nameVal = validateText(document.getElementById("nt-name").value, { max: UTOME_SEC.LIMITS.MAX_NAME, label: "Nome da tarefa" });
    const emoji = selectedTaskEmoji || "smile";
    const color = document.getElementById("nt-color").value;
    if (!time || !/^([01]\d|2[0-3]):[0-5]\d$/.test(time)) {
      alert("Horário inválido.");
      return;
    }
    if (!nameVal.ok) {
      if (!nameVal.value) document.getElementById("nt-name").focus();
      else alert(nameVal.error);
      return;
    }
    if (!newTaskDays.length) {
      alert("Selecione ao menos um dia.");
      return;
    }
    if (!COLORS.some(c => c.key === color)) {
      alert("Estado inválido.");
      return;
    }
    document.getElementById("modal-task").classList.remove("open");
    await db.collection("tasks").add({
      childId: state.activeChildId, time, name: nameVal.value, emoji, color, days: newTaskDays, active: true
    });
    logAudit("CREATE_TASK", { childId: state.activeChildId });
  });
}

async function deleteTask(taskId, taskName) {
  // I: Leaving animation
  const card = document.querySelector(`.task-card[data-id="${taskId}"]`);
  if (card) {
    card.classList.add("leaving");
    await new Promise(r => setTimeout(r, 240));
  }
  
  const removedTask = state.tasks.find(t => t.id === taskId);
  state.tasks = state.tasks.filter(t => t.id !== taskId);
  renderTaskList(null, null);

  let undone = false;

  showToast(`"${taskName || "Tarefa"}" removida`, async () => {
    undone = true;
    if (removedTask) {
      state.tasks.push(removedTask);
      state.tasks.sort((a, b) => (a.time || "").localeCompare(b.time || ""));
      renderTaskList(null, null);
    }
  });

  await new Promise(r => setTimeout(r, 4200));
  if (!undone) {
    await db.collection("tasks").doc(taskId).delete();
    logAudit("DELETE_TASK", { taskId });
  }
}

// Toast (E)
let _toastTimer = null;
function showToast(msg, onUndo) {
  const toast = document.getElementById("toast");
  const msgEl = document.getElementById("toast-msg");
  const undoBtn = document.getElementById("toast-undo");
  msgEl.textContent = msg;
  undoBtn.style.display = onUndo ? "inline" : "none";
  toast.classList.add("show");
  if (_toastTimer) clearTimeout(_toastTimer);
  _toastTimer = setTimeout(() => toast.classList.remove("show"), 4000);
  undoBtn.onclick = () => {
    toast.classList.remove("show");
    if (_toastTimer) clearTimeout(_toastTimer);
    if (onUndo) onUndo();
  };
}

  return new Promise((resolve) => {
    const modal = document.getElementById("modal-confirm");
    document.getElementById("modal-confirm-title").textContent = title;
    document.getElementById("modal-confirm-text").textContent = text;
    modal.classList.add("open");
    const onOk = () => { modal.classList.remove("open"); cleanup(); resolve(true); };
    const onCancel = () => { modal.classList.remove("open"); cleanup(); resolve(false); };
    const cleanup = () => {
      document.getElementById("modal-confirm-ok").removeEventListener("click", onOk);
      document.getElementById("modal-confirm-cancel").removeEventListener("click", onCancel);
    };
    document.getElementById("modal-confirm-ok").addEventListener("click", onOk);
    document.getElementById("modal-confirm-cancel").addEventListener("click", onCancel);
  });
}

(function setupLegalModal() {
  const tabPrivacy = document.getElementById("btn-tab-privacy");
  const tabTerms = document.getElementById("btn-tab-terms");
  const contentPrivacy = document.getElementById("content-privacy");
  const contentTerms = document.getElementById("content-terms");
  const modalLegal = document.getElementById("modal-legal");

  const selectTab = (tab) => {
    if (tab === "privacy") {
      tabPrivacy.style.background = "#fff";
      tabPrivacy.style.color = "#E55A00";
      tabTerms.style.background = "rgba(255,255,255,0.2)";
      tabTerms.style.color = "#fff";
      contentPrivacy.style.display = "block";
      contentTerms.style.display = "none";
    } else {
      tabTerms.style.background = "#fff";
      tabTerms.style.color = "#E55A00";
      tabPrivacy.style.background = "rgba(255,255,255,0.2)";
      tabPrivacy.style.color = "#fff";
      contentTerms.style.display = "block";
      contentPrivacy.style.display = "none";
    }
  };

  tabPrivacy.onclick = () => selectTab("privacy");
  tabTerms.onclick = () => selectTab("terms");

  const openLegal = (tab) => {
    selectTab(tab);
    modalLegal.style.display = "flex";
  };
  const closeLegal = () => {
    modalLegal.style.display = "none";
  };

  document.getElementById("legal-close").onclick = closeLegal;
  document.getElementById("legal-confirm").onclick = closeLegal;

  document.getElementById("open-terms").onclick = (e) => {
    e.preventDefault();
    openLegal("terms");
  };
  document.getElementById("open-privacy").onclick = (e) => {
    e.preventDefault();
    openLegal("privacy");
  };
})();
</script>
